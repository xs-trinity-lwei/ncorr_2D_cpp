cmake_minimum_required(VERSION 3.15)
project(ncorr_library LANGUAGES CXX)

# VS2019 + MSVC 2017 toolset (v141)
if(NOT MSVC)
    message(FATAL_ERROR "This CMake entry is intended for MSVC.")
endif()
if(MSVC_VERSION LESS 1910)
    message(FATAL_ERROR "MSVC 2017 (v141) or newer is required.")
endif()

set(ncorr_src
    ../src/ncorr.cpp
    ../src/Strain2D.cpp
    ../src/Disp2D.cpp
    ../src/Data2D.cpp
    ../src/ROI2D.cpp
    ../src/Image2D.cpp
    ../src/Array2D.cpp
)
set(ncorr_h
    ../include/ncorr.h
    ../include/Strain2D.h
    ../include/Disp2D.h
    ../include/Data2D.h
    ../include/ROI2D.h
    ../include/Image2D.h
    ../include/Array2D.h
)

add_library(ncorr STATIC ${ncorr_src})
set_target_properties(ncorr PROPERTIES CXX_STANDARD 11 CXX_STANDARD_REQUIRED ON CXX_EXTENSIONS OFF)
target_compile_definitions(ncorr PRIVATE NDEBUG NOMINMAX _CRT_SECURE_NO_WARNINGS _USE_MATH_DEFINES)
target_compile_options(ncorr PRIVATE /O2 /MP /permissive- /Zc:__cplusplus)
target_include_directories(ncorr PUBLIC ../include)
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ../lib)

# -----------------------------------------------------------------------------
# Bundled dependencies in repository (NO vcpkg)
# Layout is documented in docs/BUILD_VS2019_V141.md
# -----------------------------------------------------------------------------
set(THIRD_PARTY_DIR "${CMAKE_CURRENT_LIST_DIR}/../third_party" CACHE PATH "Path to bundled dependencies")
set(TP_OPENCV_DIR "${THIRD_PARTY_DIR}/opencv")
set(TP_FFTW_DIR "${THIRD_PARTY_DIR}/fftw3")
set(TP_SUITESPARSE_DIR "${THIRD_PARTY_DIR}/suitesparse")

set(TP_OPENCV_INCLUDE "${TP_OPENCV_DIR}/include")
set(TP_FFTW_INCLUDE "${TP_FFTW_DIR}/include")
set(TP_SUITESPARSE_INCLUDE "${TP_SUITESPARSE_DIR}/include")

set(TP_LIB_DIRS
    "${TP_OPENCV_DIR}/lib"
    "${TP_OPENCV_DIR}/lib/x64"
    "${TP_FFTW_DIR}/lib"
    "${TP_FFTW_DIR}/lib/x64"
    "${TP_SUITESPARSE_DIR}/lib"
    "${TP_SUITESPARSE_DIR}/lib/x64"
)

foreach(req_file
    "${TP_OPENCV_INCLUDE}/opencv2/opencv.hpp"
    "${TP_FFTW_INCLUDE}/fftw3.h"
    "${TP_SUITESPARSE_INCLUDE}/SuiteSparseQR.hpp"
)
    if(NOT EXISTS "${req_file}")
        message(FATAL_ERROR
            "Missing bundled dependency file: ${req_file}\n"
            "Please place dependencies under third_party/ as documented in docs/BUILD_VS2019_V141.md"
        )
    endif()
endforeach()

target_include_directories(ncorr PRIVATE
    "${TP_OPENCV_INCLUDE}"
    "${TP_FFTW_INCLUDE}"
    "${TP_SUITESPARSE_INCLUDE}"
)

# OpenCV: prefer opencv_world, otherwise link component libs
find_library(OPENCV_WORLD_LIB
    NAMES opencv_world opencv_world470 opencv_world480 opencv_world490
    PATHS ${TP_LIB_DIRS}
    NO_DEFAULT_PATH
)

if(OPENCV_WORLD_LIB)
    target_link_libraries(ncorr PRIVATE "${OPENCV_WORLD_LIB}")
else()
    find_library(OPENCV_CORE_LIB NAMES opencv_core opencv_core470 opencv_core480 opencv_core490 PATHS ${TP_LIB_DIRS} NO_DEFAULT_PATH)
    find_library(OPENCV_IMGPROC_LIB NAMES opencv_imgproc opencv_imgproc470 opencv_imgproc480 opencv_imgproc490 PATHS ${TP_LIB_DIRS} NO_DEFAULT_PATH)
    find_library(OPENCV_HIGHGUI_LIB NAMES opencv_highgui opencv_highgui470 opencv_highgui480 opencv_highgui490 PATHS ${TP_LIB_DIRS} NO_DEFAULT_PATH)
    find_library(OPENCV_IMGCODECS_LIB NAMES opencv_imgcodecs opencv_imgcodecs470 opencv_imgcodecs480 opencv_imgcodecs490 PATHS ${TP_LIB_DIRS} NO_DEFAULT_PATH)

    if(NOT OPENCV_CORE_LIB OR NOT OPENCV_IMGPROC_LIB OR NOT OPENCV_HIGHGUI_LIB OR NOT OPENCV_IMGCODECS_LIB)
        message(FATAL_ERROR "OpenCV libraries not found in third_party/opencv/lib[\\x64].")
    endif()

    target_link_libraries(ncorr PRIVATE
        "${OPENCV_CORE_LIB}"
        "${OPENCV_IMGPROC_LIB}"
        "${OPENCV_HIGHGUI_LIB}"
        "${OPENCV_IMGCODECS_LIB}"
    )
endif()

find_library(FFTW3_LIB NAMES fftw3 libfftw3-3 fftw3-3 PATHS ${TP_LIB_DIRS} NO_DEFAULT_PATH)
if(NOT FFTW3_LIB)
    message(FATAL_ERROR "FFTW3 library not found in third_party/fftw3/lib[\\x64].")
endif()
target_link_libraries(ncorr PRIVATE "${FFTW3_LIB}")

find_library(SPQR_LIB NAMES spqr libspqr PATHS ${TP_LIB_DIRS} NO_DEFAULT_PATH)
find_library(CHOLMOD_LIB NAMES cholmod libcholmod PATHS ${TP_LIB_DIRS} NO_DEFAULT_PATH)
find_library(AMD_LIB NAMES amd libamd PATHS ${TP_LIB_DIRS} NO_DEFAULT_PATH)
find_library(COLAMD_LIB NAMES colamd libcolamd PATHS ${TP_LIB_DIRS} NO_DEFAULT_PATH)
find_library(SUITESPARSECONFIG_LIB NAMES suitesparseconfig libsuitesparseconfig PATHS ${TP_LIB_DIRS} NO_DEFAULT_PATH)

if(NOT SPQR_LIB OR NOT CHOLMOD_LIB OR NOT AMD_LIB OR NOT COLAMD_LIB OR NOT SUITESPARSECONFIG_LIB)
    message(FATAL_ERROR "SuiteSparse libraries not found in third_party/suitesparse/lib[\\x64].")
endif()

target_link_libraries(ncorr PRIVATE
    "${SPQR_LIB}"
    "${CHOLMOD_LIB}"
    "${AMD_LIB}"
    "${COLAMD_LIB}"
    "${SUITESPARSECONFIG_LIB}"
)

install(TARGETS ncorr DESTINATION lib)
install(FILES ${ncorr_h} DESTINATION include)
